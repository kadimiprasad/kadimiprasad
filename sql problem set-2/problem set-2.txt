01. Select h.hotel_no,h.name,h.city,r.room_no,r.type,r.price from hotel h inner join room r on h.hotel_no = r.hotel_no; 
 
 
 02. Select h.hotel_no,h.name,h.city,r.room_no,r.type,r.price from hotel h inner join room r on h.hotel_no = r.hotel_no where city  = 'new york'; 

 
 03. Select name,city from guest order by city; 
 
 
 04. Select r.room_no,r.price,h.name,h.city from room r inner join hotel h on h.hotel_no = r.hotel_no where r.type = 'n' order by r.price; 
 

 05. Select city,count(*) hotel_count from hotel group by city; 
 
 
 06. Select name,city from guest group by city; 
 
 
 07. Select avg(price) from room; 
 
 
 08. Select h.name,r.room_no,r.type from room r inner join hotel h on r.hotel_no = h.hotel_no; 

 
 09. Select count(*) from booking where b.date_from like '%SEP%;
 
 
 10. Select h.name,b.date_from,b.room_no from hotel h inner join booking b on b.hotel_no = h.hotel_no where h.city = 'new york'; 
 
 
 11. Select h.name,h.city,max(r.price) high_pay from hotel h  inner join room r on h.hotel_no = r.hotel_no ; 

  
 12. Select g.name,g.city from guest g inner join on booking b join on hotel h where b.date_from like '%AUG% and h.city='new york';
 
 
 13. Select h.name,r.room_no,h.city,r.price from hotel h inner join  room r on r.hotel_no = h.hotel_no where r.price < 
     (select min(r.price) from hotel h inner join room r on h.hotel_no = r.hotel_no where h.city = 'boston'); 
 
 
 14. Select h.city,avg(r.price) avg_price from hotel h inner join room r on h.hotel_no = r.hotel_no group by h.city;
